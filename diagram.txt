class Task {
  id: string
  createdBy: string
  editedBy: string
  title: string
  description: string
  deadline: Date
  creationDate: Date
  users: string[]
  priority: Priority
  status: Status
}

  id: number
  username: string

}

class History {
  id: number

  timestamp: Date
  before: DataValue
  edition: DataValue

}

  id: number
  name: string

}

}
